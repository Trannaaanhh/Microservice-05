docker compose logs > all_logs.txt
grep -i error all_logs.txt | wc -l             # Nên: 0

## 🔴 EMERGENCY / TROUBLESHOOT

```bash
# Nếu service không khởi động
docker compose down
docker compose up --build -d

# Nếu database lỗi
docker compose exec postgres pg_isready -h postgres

# Nếu port bận
lsof -i :8000
kill -9 <PID>

# Xem chi tiết error
docker compose logs <service_name> --tail=100

# Restart một service
docker compose restart book-service

# Xóa toàn bộ dữ liệu (reset database)
docker compose down -v
docker volume prune -f

# Rebuild (không cache)
docker compose build --no-cache

# Kiểm tra kết nối giữa services
docker compose exec book-service curl -s http://order-service:8000/

# Kiểm tra RabbitMQ connection
docker compose exec order-service python -c "import pika; print('OK')"
```
